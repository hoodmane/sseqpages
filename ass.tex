\gdef\tower[#1]#2{
    \begin{scope}[#1]
    \place(0,0)
    \foreach \y in {2,...,#2} {
        \place(0, \y-1)
        \structline(0, \y-2, -1)(0, \y-1, -1)
    }
    \end{scope}
}

\gdef\etaclass(#1,#2){
    \place(#1,#2)
    \structline["\cdot\eta"](#1-1,#2-1,-1)(#1,#2,-1)
}
