    \class(0,\y)
    \structline(0,\y-1)(0,\y)

\gdef\tower[#1]#2{
    \begin{scope}[#1]
    \class(0,0)
    \foreach \y in {2,...,#2} {
        \class(0, \y-1)
        \structline(0, \y-2, -1)(0, \y-1, -1)
    }
    \end{scope}
}

\gdef\etaclass(#1,#2){
    \class(#1,#2)
    \structline["\cdot\eta"](#1-1,#2-1, -1)(#1,#2, -1)
}

\tower[xshift=7,yshift=1]{4}

\etaclass(8,2)
\class(8,3)

\etaclass(9,3)
\etaclass(9,4)
\class(9,5)

\etaclass(10,6)
\etaclass(11,7)
\class(11,6) \structline(11,6)(11,7)
\class(11,5) \structline(11,5)(11,6)

\class(14,0)

\tower[xshift=14,yshift=4]{3}
