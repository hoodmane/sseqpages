\def\tower(#1,#2){
    \foreach \y in {#2,...,25}{
        \place1(#1,\y)
        \structline(#1,\y)(#1,\y+1)
    }
}

\def\towergroup(#1,#2){
    \tower(#1,#2)
    \place1(#1+1,#2+1)
    \place1(#1+3,#2+1)

    \structline(#1,#2+1)(#1+3,#2+2)
    \structline(#1,#2+2)(#1+3,#2+3)
}
